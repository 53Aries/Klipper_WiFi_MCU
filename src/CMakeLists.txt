# src/CMakeLists.txt - ESP-IDF main component for KlipperWifiMcu
#
# PlatformIO with ESP-IDF framework ignores build_src_filter. Source file
# selection is done here via the BUILD_DIR cmake variable, which is set by
# build_properties.temp.cmake and IS available in the
# component_get_requirements.cmake subprocess.
#
# BUILD_DIR contains the full build path, e.g.:
#   "C:/KlipperWifiMcu/.pio/build/host_esp"  (host_esp env)
#   "C:/KlipperWifiMcu/.pio/build/mcu_esp"   (mcu_esp env)

cmake_minimum_required(VERSION 3.16)

# Shared protocol header lives one level up at include/kwm_protocol.h
set(KWM_INCLUDE_DIR "${CMAKE_CURRENT_SOURCE_DIR}/../include")

# BUILD_DIR is injected by build_properties.temp.cmake (generated by PlatformIO)
# and IS available in the requirements subprocess — unlike cmake cache variables
# and $ENV{PIOENV} which are both empty in that subprocess.
if("${BUILD_DIR}" MATCHES "/host_esp")
    idf_component_register(
        SRCS
            "host_esp/main.c"
            "host_esp/kwm_spi.c"
            "host_esp/wifi_ap.c"
            "host_esp/tcp_server.c"
            "host_esp/bridge.c"
        INCLUDE_DIRS
            "host_esp"
            "${KWM_INCLUDE_DIR}"
    )

elseif("${BUILD_DIR}" MATCHES "/mcu_esp")
    idf_component_register(
        SRCS
            "mcu_esp/main.c"
            "mcu_esp/kwm_uart.c"
            "mcu_esp/wifi_sta.c"
            "mcu_esp/tcp_client.c"
            "mcu_esp/bridge.c"
        INCLUDE_DIRS
            "mcu_esp"
            "${KWM_INCLUDE_DIR}"
    )

else()
    message(FATAL_ERROR
        "Cannot determine firmware target from BUILD_DIR='${BUILD_DIR}'. "
        "Expected path containing '/host_esp' or '/mcu_esp'.")
endif()
